# Код 1С для интеграции с Lakshmi

## Общий модуль: `ЛакшмиИнтеграция`

Все запросы к API Lakshmi используют:
- Заголовок `X-Api-Key` с ключом интеграции
- Формат JSON
- Base URL сервера (например `https://your-domain.tld`)

---

## 1. Общие функции

```bsl
// Общий модуль "ЛакшмиИнтеграция"

Перем БазовыйURL;   // "https://your-domain.tld"
Перем КлючAPI;       // значение ONEC_API_KEY

Процедура УстановитьНастройки() Экспорт
    БазовыйURL = Константы.ЛакшмиURL.Получить();
    КлючAPI     = Константы.ЛакшмиAPIKey.Получить();
КонецПроцедуры

// Выполнить HTTP-запрос к API Lakshmi
//
// Параметры:
//   Метод   - Строка - "GET" или "POST"
//   Путь    - Строка - путь endpoint, например "/onec/health"
//   Тело    - Строка - JSON-строка тела запроса (для POST), "" для GET
//   ДопЗаголовки - Соответствие - дополнительные заголовки (опционально)
//
// Возвращаемое значение:
//   Структура - КодОтвета (Число), Тело (Строка)
//
Функция ВыполнитьЗапрос(Метод, Путь, Тело = "", ДопЗаголовки = Неопределено) Экспорт

    УстановитьНастройки();

    СтруктураURL = РазобратьURL(БазовыйURL);
    Хост   = СтруктураURL.Хост;
    Порт   = СтруктураURL.Порт;
    Защита = СтруктураURL.Защита;

    Соединение = Новый HTTPСоединение(Хост, Порт,,,, 30, Защита);

    Запрос = Новый HTTPЗапрос(Путь);
    Запрос.Заголовки.Вставить("X-Api-Key", КлючAPI);
    Запрос.Заголовки.Вставить("Content-Type", "application/json");

    Если ДопЗаголовки <> Неопределено Тогда
        Для Каждого Заголовок Из ДопЗаголовки Цикл
            Запрос.Заголовки.Вставить(Заголовок.Ключ, Заголовок.Значение);
        КонецЦикла;
    КонецЕсли;

    Если НЕ ПустаяСтрока(Тело) Тогда
        Запрос.УстановитьТелоИзСтроки(Тело, КодировкаТекста.UTF8);
    КонецЕсли;

    Если ВРег(Метод) = "POST" Тогда
        Ответ = Соединение.ОтправитьДляОбработки(Запрос);
    Иначе
        Ответ = Соединение.Получить(Запрос);
    КонецЕсли;

    Результат = Новый Структура;
    Результат.Вставить("КодОтвета", Ответ.КодСостояния);
    Результат.Вставить("Тело", Ответ.ПолучитьТелоКакСтроку(КодировкаТекста.UTF8));

    Возврат Результат;

КонецФункции

Функция РазобратьURL(URL)
    Результат = Новый Структура("Хост, Порт, Защита");
    Если СтрНачинаетсяС(URL, "https://") Тогда
        Результат.Защита = Новый ЗащищенноеСоединениеOpenSSL();
        Результат.Порт = 443;
        Результат.Хост = СтрЗаменить(URL, "https://", "");
    Иначе
        Результат.Защита = Неопределено;
        Результат.Порт = 80;
        Результат.Хост = СтрЗаменить(URL, "http://", "");
    КонецЕсли;
    // Убрать trailing slash
    Если СтрЗаканчиваетсяНа(Результат.Хост, "/") Тогда
        Результат.Хост = Лев(Результат.Хост, СтрДлина(Результат.Хост) - 1);
    КонецЕсли;
    Возврат Результат;
КонецФункции

Функция РазобратьJSON(СтрокаJSON) Экспорт
    ЧтениеJSON = Новый ЧтениеJSON;
    ЧтениеJSON.УстановитьСтроку(СтрокаJSON);
    Возврат ПрочитатьJSON(ЧтениеJSON);
КонецФункции

Функция СформироватьJSON(Данные) Экспорт
    ЗаписьJSON = Новый ЗаписьJSON;
    ЗаписьJSON.УстановитьСтроку();
    ЗаписатьJSON(ЗаписьJSON, Данные);
    Возврат ЗаписьJSON.Закрыть();
КонецФункции

Функция НовыйUUID() Экспорт
    Возврат Строка(Новый УникальныйИдентификатор);
КонецФункции
```

## 2. Проверка соединения (Health Check)

```bsl
// POST /onec/health
// Проверка связи с сервером Lakshmi
//
// Возвращаемое значение:
//   Булево - Истина если сервер доступен
//
Функция ПроверитьСоединение() Экспорт

    Попытка
        Результат = ВыполнитьЗапрос("POST", "/onec/health");
    Исключение
        Сообщить("Ошибка соединения с Lakshmi: " + ОписаниеОшибки());
        Возврат Ложь;
    КонецПопытки;

    Если Результат.КодОтвета = 200 Тогда
        Данные = РазобратьJSON(Результат.Тело);
        Возврат Данные.status = "ok";
    Иначе
        Сообщить("Lakshmi health check failed: HTTP " + Результат.КодОтвета);
        Возврат Ложь;
    КонецЕсли;

КонецФункции
```

## 3. Загрузка чека (Receipt)

```bsl
// POST /onec/receipt
// Отправка чека с покупкой в Lakshmi для начисления бонусов
//
// Параметры:
//   ДанныеЧека - Структура:
//     .НомерЧека       - Строка - уникальный GUID чека
//     .Дата             - Дата
//     .КодМагазина      - Строка
//     .TelegramID       - Число - telegram_id клиента (0 если неизвестен)
//     .GUIDКлиента1С    - Строка - GUID клиента в 1С (пусто если неизвестен)
//     .Позиции          - Массив структур:
//       .КодТовара       - Строка
//       .Наименование    - Строка
//       .Количество      - Число
//       .Цена            - Число
//       .СуммаСкидки     - Число
//       .Акционный       - Булево
//       .НомерСтроки     - Число (уникальный в пределах чека)
//       .НачисленоБонусов - Число
//     .СуммаИтого        - Число
//     .СуммаСкидок       - Число
//     .СписаноБонусов    - Число
//     .НачисленоБонусов  - Число
//
// Возвращаемое значение:
//   Структура - результат или Неопределено при ошибке
//
Функция ОтправитьЧек(ДанныеЧека) Экспорт

    // Формируем позиции
    Позиции = Новый Массив;
    Для Каждого Строка Из ДанныеЧека.Позиции Цикл
        Позиция = Новый Структура;
        Позиция.Вставить("product_code",     Строка.КодТовара);
        Позиция.Вставить("name",             Строка.Наименование);
        Позиция.Вставить("quantity",         Формат(Строка.Количество, "ЧРД=.; ЧН=0; ЧГ="));
        Позиция.Вставить("price",            Формат(Строка.Цена, "ЧРД=.; ЧН=0.00; ЧГ="));
        Позиция.Вставить("discount_amount",  Формат(Строка.СуммаСкидки, "ЧРД=.; ЧН=0.00; ЧГ="));
        Позиция.Вставить("is_promotional",   Строка.Акционный);
        Позиция.Вставить("line_number",      Строка.НомерСтроки);
        Позиция.Вставить("bonus_earned",     Формат(Строка.НачисленоБонусов, "ЧРД=.; ЧН=0.00; ЧГ="));
        Позиции.Добавить(Позиция);
    КонецЦикла;

    // Формируем клиента
    Клиент = Новый Структура;
    Если ДанныеЧека.TelegramID > 0 Тогда
        Клиент.Вставить("telegram_id", ДанныеЧека.TelegramID);
    КонецЕсли;
    Если НЕ ПустаяСтрока(ДанныеЧека.GUIDКлиента1С) Тогда
        Клиент.Вставить("one_c_guid", ДанныеЧека.GUIDКлиента1С);
    КонецЕсли;

    // Формируем итоги
    Итоги = Новый Структура;
    Итоги.Вставить("total_amount",    Формат(ДанныеЧека.СуммаИтого, "ЧРД=.; ЧН=0.00; ЧГ="));
    Итоги.Вставить("discount_total",  Формат(ДанныеЧека.СуммаСкидок, "ЧРД=.; ЧН=0.00; ЧГ="));
    Итоги.Вставить("bonus_spent",     Формат(ДанныеЧека.СписаноБонусов, "ЧРД=.; ЧН=0.00; ЧГ="));
    Итоги.Вставить("bonus_earned",    Формат(ДанныеЧека.НачисленоБонусов, "ЧРД=.; ЧН=0.00; ЧГ="));

    // Собираем тело запроса
    ТелоЗапроса = Новый Структура;
    ТелоЗапроса.Вставить("receipt_guid", ДанныеЧека.НомерЧека);
    ТелоЗапроса.Вставить("datetime",    Формат(ДанныеЧека.Дата, "ДФ=yyyy-MM-ddTHH:mm:ss"));
    ТелоЗапроса.Вставить("store_id",    ДанныеЧека.КодМагазина);
    ТелоЗапроса.Вставить("customer",    Клиент);
    ТелоЗапроса.Вставить("positions",   Позиции);
    ТелоЗапроса.Вставить("totals",      Итоги);

    // Заголовок идемпотентности
    ДопЗаголовки = Новый Соответствие;
    ДопЗаголовки.Вставить("X-Idempotency-Key", НовыйUUID());

    Попытка
        Результат = ВыполнитьЗапрос("POST", "/onec/receipt", СформироватьJSON(ТелоЗапроса), ДопЗаголовки);
    Исключение
        Сообщить("Ошибка отправки чека: " + ОписаниеОшибки());
        Возврат Неопределено;
    КонецПопытки;

    Если Результат.КодОтвета = 200 ИЛИ Результат.КодОтвета = 201 Тогда
        Возврат РазобратьJSON(Результат.Тело);
    Иначе
        Сообщить("Ошибка отправки чека: HTTP " + Результат.КодОтвета + " — " + Результат.Тело);
        Возврат Неопределено;
    КонецЕсли;

КонецФункции
```

## 4. Синхронизация клиента

```bsl
// POST /onec/customer
// Синхронизация данных клиента (обновление бонусов, привязка GUID и т.д.)
//
// Параметры:
//   TelegramID     - Число - telegram_id клиента (0 если неизвестен)
//   QRКод          - Строка - QR-код клиента (пусто если неизвестен)
//   GUID1С         - Строка - GUID клиента в 1С
//   БонусныйБаланс - Число - текущий баланс бонусов
//

//   Структура - данные клиента или Неопределено при ошибке

    Если TelegramID > 0 Тогда
        ТелоЗапроса.Вставить("telegram_id", TelegramID);
    КонецЕсли;
    Если НЕ ПустаяСтрока(QRКод) Тогда
        ТелоЗапроса.Вставить("qr_code", QRКод);
    КонецЕсли;
    Если НЕ ПустаяСтрока(GUID1С) Тогда
        ТелоЗапроса.Вставить("one_c_guid", GUID1С);
    КонецЕсли;
    Если БонусныйБаланс > 0 Тогда
        ТелоЗапроса.Вставить("bonus_balance", Формат(БонусныйБаланс, "ЧРД=.; ЧН=0.00; ЧГ="));
    КонецЕсли;

    Попытка
        Результат = ВыполнитьЗапрос("POST", "/onec/customer", СформироватьJSON(ТелоЗапроса));
    Исключение
        Сообщить("Ошибка синхронизации клиента: " + ОписаниеОшибки());
        Возврат Неопределено;
    КонецПопытки;

    Если Результат.КодОтвета = 200 Тогда
        Возврат РазобратьJSON(Результат.Тело);
    Иначе
        Сообщить("Ошибка синхронизации клиента: HTTP " + Результат.КодОтвета + " — " + Результат.Тело);
        Возврат Неопределено;
    КонецЕсли;
```

## 7. Получение новых заказов

```bsl
// GET /onec/orders/pending
// Получить заказы со статусом "new" для обработки в 1С
// ВАЖНО: после получения статус заказов автоматически меняется на "assembly"
//
// Параметры:
//   ПослеДаты - Дата - получить заказы созданные после этой даты (опционально)
//   Лимит     - Число - максимальное количество (по умолчанию 50, макс 200)
//
// Возвращаемое значение:
//   Массив структур с заказами или Неопределено при ошибке
//
Функция ПолучитьНовыеЗаказы(ПослеДаты = Неопределено, Лимит = 50) Экспорт

    Путь = "/onec/orders/pending?limit=" + Формат(Лимит, "ЧН=50; ЧГ=");

    Если ПослеДаты <> Неопределено Тогда
        Путь = Путь + "&after=" + Формат(ПослеДаты, "ДФ=yyyy-MM-ddTHH:mm:ss");
    КонецЕсли;

    Попытка
        Результат = ВыполнитьЗапрос("GET", Путь);
    Исключение
        Сообщить("Ошибка получения заказов: " + ОписаниеОшибки());
        Возврат Неопределено;
    КонецПопытки;

    Если Результат.КодОтвета = 200 Тогда
        Данные = РазобратьJSON(Результат.Тело);
        Возврат Данные.orders;
    Иначе
        Сообщить("Ошибка получения заказов: HTTP " + Результат.КодОтвета + " — " + Результат.Тело);
        Возврат Неопределено;
    КонецЕсли;

КонецФункции
```

**Структура каждого заказа в ответе:**

```
order_id          - Число   - ID заказа
created_at        - Строка  - дата создания ISO
status            - Строка  - "new" (до получения), "assembly" (после)
payment_method    - Строка  - "online" / "cash" / "card"
fulfillment_type  - Строка  - "delivery" / "pickup"
customer          - Структура:
    .id           - Число
    .telegram_id  - Число
    .email        - Строка
    .full_name    - Строка
    .phone        - Строка
delivery          - Структура:
    .address      - Строка
    .comment      - Строка
prices            - Структура:
    .products_price  - Строка (Decimal)
    .delivery_price  - Строка (Decimal)
    .total_price     - Строка (Decimal)
items             - Массив:
    .product_code - Строка
    .name         - Строка
    .quantity     - Число
    .price        - Строка (Decimal)
```

## 8. Обновление статуса заказа

```bsl
// POST /onec/order/status
// Обновить статус заказа в Lakshmi
//
// Допустимые переходы статусов:
//   new       → accepted, canceled
//   accepted  → assembly, canceled, new
//   assembly  → ready, canceled
//   ready     → delivery, completed, canceled
//   delivery  → arrived, canceled
//   arrived   → completed, canceled
//   canceled  → new
//
// Параметры:
//   IDЗаказа      - Число - ID заказа в Lakshmi
//   НовыйСтатус   - Строка - целевой статус
//   GUID1С        - Строка - GUID заказа в 1С (опционально, привязывается при первом вызове)
//   ПричинаОтмены - Строка - причина (при статусе "canceled")
//
// Возвращаемое значение:
//   Структура - результат или Неопределено при ошибке
//
Функция ОбновитьСтатусЗаказа(IDЗаказа, НовыйСтатус, GUID1С = "", ПричинаОтмены = "") Экспорт

    ТелоЗапроса = Новый Структура;
```

## 10. Входящие запросы ОТ Lakshmi В 1С

Lakshmi отправляет в 1С два типа запросов. 1С должна принимать их на своём HTTP-сервисе.

### 10.1 Новый заказ от Lakshmi

Lakshmi отправляет `POST` на URL из настройки `ONEC_ORDER_URL` с заголовками:
- `X-Api-Key: {INTEGRATION_API_KEY}`
- `X-Idempotency-Key: {UUID}`
- `Content-Type: application/json`

**Тело запроса (JSON):**
```json
{
  "order_id": 123,
  "created_at": "2026-03-10T10:30:00",
  "customer": {
    "id": 1,
    "telegram_id": 123456789,
    "email": "user@example.com",
    "phone": "+79991234567",
    "full_name": "Иванов Иван"
  },
  "delivery": {
    "type": "delivery",
    "address": "ул. Пушкина, д. 10",
    "comment": "Позвонить перед доставкой"
  },
  "payment_method": "online",
  "payment_id": "2d4f5a6b-...",
  "payment_status": "authorized",
  "fulfillment_type": "delivery",
  "prices": {
    "products_price": "990.00",
    "delivery_price": "50.00",
    "total_price": "1040.00"
  },
  "store_id": "store123",
  "items": [
    {
      "product_code": "SKU123",
      "name": "Молоко 1л",
      "quantity": 2,
      "price": "495.00",
      "line_total": "990.00"
    }
  ]
}
```

**1С должна вернуть (200 или 201):**
```json
{
  "onec_guid": "GUID-заказа-в-1С"
}
```

### 10.2 Уведомление об отмене заказа

Lakshmi отправляет `POST` на тот же `ONEC_ORDER_URL`:

**Тело запроса:**
```json
{
  "order_id": 123,
  "status": "canceled",
  "cancel_reason": "Клиент отменил",
  "canceled_by": "customer",
  "payment_method": "online",
  "payment_id": "2d4f5a6b-...",
  "payment_result": "hold_canceled",
  "courier_telegram_id": null
}
```

**1С должна вернуть:**
```json
{
  "status": "ok"
}
```

## Пример обработчика HTTP-сервиса 1С для приёма заказов

```bsl
// Обработка входящего POST-запроса от Lakshmi
// Метод HTTP-сервиса 1С

Функция ПолучитьЗаказОтЛакшми(Запрос)

    // 1. Проверить API-ключ
    КлючИзЗапроса = Запрос.Заголовки.Получить("X-Api-Key");
    Если КлючИзЗапроса <> Константы.ЛакшмиAPIKey.Получить() Тогда
        Ответ = Новый HTTPСервисОтвет(401);
        Ответ.УстановитьТелоИзСтроки("{""detail"": ""Bad API key""}");
        Возврат Ответ;
    КонецЕсли;

    // 2. Разобрать JSON
    ТелоСтрока = Запрос.ПолучитьТелоКакСтроку(КодировкаТекста.UTF8);
    ЧтениеJSON = Новый ЧтениеJSON;
    ЧтениеJSON.УстановитьСтроку(ТелоСтрока);
    Данные = ПрочитатьJSON(ЧтениеJSON);

    // 3. Проверить — это отмена или новый заказ?
    Если Данные.Свойство("status") И Данные.status = "canceled" Тогда
        // Обработка отмены
        ОбработатьОтменуЗаказа(Данные);
        Ответ = Новый HTTPСервисОтвет(200);
        Ответ.УстановитьТелоИзСтроки("{""status"": ""ok""}");
        Возврат Ответ;
    КонецЕсли;

    // 4. Проверить идемпотентность
    КлючИдемпотентности = Запрос.Заголовки.Получить("X-Idempotency-Key");
    // ... проверка дублей по ключу ...

    // 5. Создать заказ в 1С
    GUIDЗаказа = СоздатьЗаказВ1С(Данные);

    // 6. Вернуть GUID
    РезультатJSON = Новый Структура;
    РезультатJSON.Вставить("onec_guid", GUIDЗаказа);

    ЗаписьJSON = Новый ЗаписьJSON;
    ЗаписьJSON.УстановитьСтроку();
    ЗаписатьJSON(ЗаписьJSON, РезультатJSON);

    Ответ = Новый HTTPСервисОтвет(201);
    Ответ.Заголовки.Вставить("Content-Type", "application/json");
    Ответ.УстановитьТелоИзСтроки(ЗаписьJSON.Закрыть());
    Возврат Ответ;

КонецФункции
```
